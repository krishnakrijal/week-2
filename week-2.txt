CREATE function [dbo].[CalculateAge](@DateOfBirth datetime)
returns int
as begin
  declare @age int;

   set @age = DATEDIFF(year,@DateOfBirth,GETDATE())
     -CASE 
	 WHEN CONVERT(DATE,@DateOfBirth)>CONVERT(DATE,GETDATE())
	 THEN 1
	 ELSE 0
	 END;

	 return @age
	 end ;

GO

CREATE view [dbo].[CustomerSales]
as
select c.Id as 'Customer Id', c.FirstName, c.LastName,CONCAT(c.FirstName,' ',c.LastName) as 'Full Name',
s.DateSold, p.[Name], p.Price,[dbo].[GetCustomerAmount](c.Id) as 'total purchases'

	from Customer c
		left join 
			Sale s on 
				c.Id = s.CustomerId
					left join 
						Product p on
							s.ProductId = p.Id 

GO
